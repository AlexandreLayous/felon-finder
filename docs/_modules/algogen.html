def one_point_crossover(parent1, parent2):
    """
    A point is randomly picked on both parent's arrays. Data beyond this point is swapped.

    Args : 2 arrays representing parent 1 and 2

    Returns : 2 arrays of children
    """
    parent1_tensor = tf.constant(parent1)
    parent2_tensor = tf.constant(parent2)

    child1 = tf.TensorArray(dtype=parent1_tensor.dtype, size=tf.size(parent1_tensor))
    child2 = tf.TensorArray(dtype=parent1_tensor.dtype, size=tf.size(parent2_tensor))

    departure_point = random.randint(0, tf.size(parent1_tensor))


    for i in range(departure_point):
        child1 = child1.write(i, parent1_tensor[i])
        child2 = child2.write(i, parent2_tensor[i])

    for i in range(departure_point, tf.size(parent2_tensor)):
        child1 = child1.write(i, parent2_tensor[i])
        child2 = child2.write(i, parent1_tensor[i])

    return child1.stack()

def swap_mutation(parent, num_bits):
    # Générer num_bits points de mutation uniques
    points = tf.random.shuffle(tf.range(1, tf.shape(parent)[0]))[:num_bits]

    # Extraire les valeurs des points de mutation
    values = tf.gather(parent, points)

    # Permuter les valeurs
    permuted_values = tf.random.shuffle(values)

    # Mettre à jour les valeurs des points de mutation dans l'enfant
    child = tf.tensor_scatter_nd_update(parent, tf.expand_dims(points, axis=1), permuted_values)

    return child

def gaussian_noise_1(parent, mean, std, n):
    """
    X points are randomly picked on both parent's arrays. Data beyond this point is swapped.

    Args : 2 arrays representing parent 1 and 2 and the number of points

    Returns : 2 arrays of children

    """
    p = parent.numpy()
    noise = np.random.normal(mean, std, len(p))
    points = sorted(random.sample(range(len(p)), n))
    child = p[:]

    for i in range(len(p)):
        if i in points:
            child[i] += noise[i]
    return child

def mean_parents(parent1, parent2):
    """
    Calcule la moyenne des deux parents.parent2_tensor, parent1_tensor,parent2_tensor, parent1_tensor, parent2_tensor

    Args:
        Les 2 parents sélectionnés par l'utilisateur.

    Returns:
        Un enfant (liste numpy) dont les valeurs sont la moyenne des valeurs de chaque parent.
    """

    p1 = parent1.numpy()
    p2 = parent2.numpy()
    child = []

    for i in range(len(p1)):
        child.append((p1[i]+p2[i])/2)

    return child

def mean_parents1(*parents):
    """
   Calcule la moyenne des deux ou plusieurs parents.

   Args:
       N parents sélectionnés par l'utilisateur.

   Returns:
       Un enfant (liste numpy) dont les valeurs sont la moyenne pondérée des valeurs de chaque parent.
   """
    num_parents = len(parents)
    child = np.zeros_like(parents[0].numpy())

    for parent in parents:
        p = parent.numpy()
        child += p

    child /= num_parents

    return child

def one_selection(parent, std, m):
    """
   Dans le cas où l'utilisateur sélectionne un parent.

   Args:
       Le vecteur encodé du parent sélectionné
   Returns:
      4 outputs (vecteurs) où nous avons introduits du bruit gaussien prêt à être décodés
   """

    output1 = gaussian_noise_1(parent, m, std+1, 30)
    output2 = gaussian_noise_1(parent, m, std+2, 35)/2
    output3 = gaussian_noise_1(parent, m, std+3, 35)/5
    output4 = gaussian_noise_1(parent, m, std+4, 30)/3

    x_modified = []
    x_modified.append(output1)
    x_modified.append(output2)
    x_modified.append(output3)
    x_modified.append(output4)

    x_to_decode = tf.stack(x_modified, axis = 0)

    return x_to_decode

def multiple_selections(*parents):
    
    outputs = []

    for parent1, parent2 in zip(parents[:-1], parents[1:]):
        output1 = several_points_crossover_v2(parent1, parent2)
        output2 = one_point_crossover(parent1, parent2)
        output3 = mean_parents(parent1, parent2)
        output4 = several_points_crossover(parent1, parent2, 7)

        outputs.extend([output1, output2, output3, output4])

    # Sélectionner 3 éléments aléatoires de la liste 'outputs' + moyenne de tous les parents
    selected_outputs = random.sample(outputs, 3)
    selected_outputs.append(mean_parents1(*parents))


    x_to_decode = tf.stack(selected_outputs, axis=0)

    return x_to_decode
